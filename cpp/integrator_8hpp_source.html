45/**
46 * @brief Generalized-alpha integrator
47 *
48 * This class handles the time integration scheme for solving 0D blood
49 * flow system.
50 *
51 * Flow rate, pressure, and other hemodynamic quantities in 0D models of
52 * vascular anatomies are governed by a system of nonlinear
53 * differential-algebraic equations (DAEs):
54 *
55 * \f[
56 * \mathbf{E}(\mathbf{y}, t) \cdot \dot{\mathbf{y}}+\mathbf{F}(\mathbf{y}, t)
57 * \cdot \mathbf{y}+\mathbf{c}(\mathbf{y}, t)=\mathbf{0} \f]
58 *
59 * Here, \f$y\f$ is the vector of solution quantities and \f$\dot{y}\f$ is the
60 * time derivative of \f$y\f$. \f$N\f$ is the total number of equations and the
61 * total number of global unknowns. The DAE system is solved implicitly using
62 * the generalized-\f$\alpha\f$ method \cite JANSEN2000305.
63 *
64 * @tparam T Scalar type (e.g. `float`, `double`)
65 * `ALGEBRA::SparseSystem`)
66 */
67template <typename T>
68class Integrator {
69 private:
70 T alpha_m;
71 T alpha_f;
72 T alpha_m_inv;
73 T alpha_f_inv;
74 T gamma;
75 T gamma_inv;
76 T time_step_size;
77 T time_step_size_inv;
78 T y_dot_coeff;
79 T atol;
80 T y_init_coeff;
81 T ydot_init_coeff;
82 int max_iter;
83 int size;
84 Eigen::Matrix<T, Eigen::Dynamic, 1> y_af;
85 Eigen::Matrix<T, Eigen::Dynamic, 1> ydot_am;

138template <typename T>
139Integrator<T>::Integrator(MODEL::Model<T>* model, T time_step_size, T rho,
140 T atol, int max_iter) {
141 this->model = model;
142 alpha_m = 0.5 * (3.0 - rho) / (1.0 + rho);
143 alpha_f = 1.0 / (1.0 + rho);
144 alpha_m_inv = 1.0 / alpha_m;
145 alpha_f_inv = 1.0 / alpha_f;
146 gamma = 0.5 + alpha_m - alpha_f;
147 gamma_inv = 1.0 / gamma;
148
149 y_init_coeff = alpha_f * 0.5 * time_step_size;
150 ydot_init_coeff = (1.0 + alpha_m * ((gamma - 0.5) * gamma_inv - 1.0));
151
152 size = model->dofhandler.size();
153 system = SparseSystem<T>(size);
154 this->time_step_size = time_step_size;
155 this->atol = atol;
156 this->max_iter = max_iter;
157 time_step_size_inv = 1.0 / time_step_size;
158
159 y_af = Eigen::Matrix<T, Eigen::Dynamic, 1>(size);
160 ydot_am = Eigen::Matrix<T, Eigen::Dynamic, 1>(size);
161
162 y_dot_coeff = alpha_m * alpha_f_inv * gamma_inv * time_step_size_inv;
163
164 // Make some memory reservations
165 system.reserve(model);
166}

181template <typename T>
182void Integrator<T>::update_params(T time_step_size) {
183 y_init_coeff = alpha_f * 0.5 * time_step_size;
184 this->time_step_size = time_step_size;
185 time_step_size_inv = 1.0 / time_step_size;
186 y_dot_coeff = alpha_m * alpha_f_inv * gamma_inv * time_step_size_inv;
187 model->update_constant(system);
188 model->update_time(system, 0.0);
189}

191template <typename T>
192State<T> Integrator<T>::step(State<T>& old_state, T time) {
193 // Predictor + initiator step
194 y_af.setZero();
195 ydot_am.setZero();
196 y_af += old_state.y + old_state.ydot * y_init_coeff;
197 ydot_am += old_state.ydot * ydot_init_coeff;
198
199 // Determine new time
200 T new_time = time + alpha_f * time_step_size;
201
202 // Update time-dependent element contributions in system
203 model->update_time(system, new_time);
204
205 for (size_t i = 0; i < max_iter; i++) {
206 // Update solution-dependent element contribitions
207 model->update_solution(system, y_af, ydot_am);
208
209 // Update residuum and check termination criteria
210 system.update_residual(y_af, ydot_am);
211 if (system.residual.cwiseAbs().maxCoeff() < atol) {
212 break;
213 }
214
215 // Abort if maximum number of non-linear iterations is reached
216 else if (i == max_iter - 1) {
217 throw std::runtime_error(
218 "Maximum number of non-linear iterations reached.");
219 }
220
221 // Determine jacobian
222 system.update_jacobian(y_dot_coeff);
223
224 // Solve system
225 system.solve();
226
227 // Add increment to solution
228 y_af += system.dy;
229 ydot_am += system.dy * y_dot_coeff;
230 }
231
232 // Set new state
233 State<T> new_state = State<T>::Zero(size);
234 new_state.y += old_state.y + (y_af - old_state.y) * alpha_f_inv;
235 new_state.ydot += old_state.ydot + (ydot_am - old_state.ydot) * alpha_m_inv;
236
237 return new_state;
238}
