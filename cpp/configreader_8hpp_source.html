74template <typename T>
75struct SimulationParameters {
76 // Negative value indicates this has not
77 // been read from config file yet.
78 T sim_time_step_size; ///< Simulation time step size
79 T sim_abs_tol; ///< Absolute tolerance for simulation
80
81 int sim_num_cycles; ///< Number of cardiac cycles to simulate
82 int sim_pts_per_cycle; ///< Number of time steps per cardiac cycle
83 int sim_num_time_steps; ///< Total number of time steps
84 int sim_nliter; ///< Maximum number of non-linear iterations in time
85 ///< integration
86 int output_interval; ///< Interval of writing output
87
88 bool sim_steady_initial; ///< Start from steady solution
89 bool
90 output_variable_based; ///< Output variable based instead of vessel based
91 bool output_mean_only; ///< Output only the mean value
92 bool output_derivative; ///< Output derivatives
93 bool output_all_cycles; ///< Output all cardiac cycles
94
95 bool sim_coupled; ///< Running 0D simulation coupled with external solver
96 T sim_external_step_size; ///< Step size of external solver if running
97 ///< coupled
98};

100/**
101 * @brief Load the simulation parameters from a json configuration
102 *
103 * @tparam T Scalar type (e.g. `float`, `double`)
104 * @param config The json configuration
105 * @return SimulationParameters<T> Simulation parameters read from configuration
106 */
107template <typename T>
108SimulationParameters<T> load_simulation_params(const nlohmann::json& config) {
109 DEBUG_MSG("Loading simulation parameters");
110 SimulationParameters<T> sim_params;
111 const auto& sim_config = config["simulation_parameters"];
112 sim_params.sim_coupled = sim_config.value("coupled_simulation", false);
113 if (!sim_params.sim_coupled) {
114 sim_params.sim_num_cycles = sim_config["number_of_cardiac_cycles"];
115 sim_params.sim_pts_per_cycle =
116 sim_config["number_of_time_pts_per_cardiac_cycle"];
117 sim_params.sim_num_time_steps =
118 (sim_params.sim_pts_per_cycle - 1) * sim_params.sim_num_cycles + 1;
119 sim_params.sim_external_step_size = 0.0;
120 } else {
121 sim_params.sim_num_cycles = 1;
122 sim_params.sim_num_time_steps = sim_config["number_of_time_pts"];
123 sim_params.sim_pts_per_cycle = sim_params.sim_num_time_steps;
124 sim_params.sim_external_step_size =
125 sim_config.value("external_step_size", 0.1);
126 }
127 sim_params.sim_abs_tol = sim_config.value("absolute_tolerance", 1e-8);
128 sim_params.sim_nliter = sim_config.value("maximum_nonlinear_iterations", 30);
129 sim_params.sim_steady_initial = sim_config.value("steady_initial", true);
130 sim_params.output_variable_based =
131 sim_config.value("output_variable_based", false);
132 sim_params.output_interval = sim_config.value("output_interval", 1);
133 sim_params.output_mean_only = sim_config.value("output_mean_only", false);
134 sim_params.output_derivative = sim_config.value("output_derivative", false);
135 sim_params.output_all_cycles = sim_config.value("output_all_cycles", false);
136 DEBUG_MSG("Finished loading simulation parameters");
137 return sim_params;
138}

140/**
141 * @brief Load model from a configuration
142 *
143 * @tparam T Scalar type (e.g. `float`, `double`)
144 * @param config The json configuration
145 */
146template <typename T>
147void load_simulation_model(const nlohmann::json& config,
148 MODEL::Model<T>& model) {
149 DEBUG_MSG("Loading model");
150
151 // Create list to store block connections while generating blocks
152 std::vector<std::tuple<std::string, std::string>> connections;
153 // Create vessels
154 DEBUG_MSG("Load vessels");
155 std::map<int, std::string> vessel_id_map;
156 const auto& vessels = config["vessels"];
157 for (size_t i = 0; i < vessels.size(); i++) {
158 const auto& vessel_config = vessels[i];
159 const auto& vessel_values = vessel_config["zero_d_element_values"];
160 const std::string vessel_name = vessel_config["vessel_name"];
161 vessel_id_map.insert({vessel_config["vessel_id"], vessel_name});
162 if (vessel_config["zero_d_element_type"] == "BloodVessel") {
163 model.add_block(MODEL::BlockType::BLOODVESSEL,
164 {model.add_parameter(vessel_values["R_poiseuille"]),
165 model.add_parameter(vessel_values.value("C", 0.0)),
166 model.add_parameter(vessel_values.value("L", 0.0)),
167 model.add_parameter(
168 vessel_values.value("stenosis_coefficient", 0.0))},
169 vessel_name);
170 DEBUG_MSG("Created vessel " << vessel_name);
171 } else {
172 throw std::invalid_argument("Unknown vessel type");
173 }
174
175 // Read connected boundary conditions
176 if (vessel_config.contains("boundary_conditions")) {
177 const auto& vessel_bc_config = vessel_config["boundary_conditions"];
178 if (vessel_bc_config.contains("inlet")) {
179 connections.push_back({vessel_bc_config["inlet"], vessel_name});
180 }
181 if (vessel_bc_config.contains("outlet")) {
182 connections.push_back({vessel_name, vessel_bc_config["outlet"]});
183 }
184 }
185 }
186
187 // Create map for boundary conditions to boundary condition type
188 DEBUG_MSG("Create BC name to BC type map");
189 const auto& bc_configs = config["boundary_conditions"];
190 std::map<std::string, std::string> bc_type_map;
191 for (size_t i = 0; i < bc_configs.size(); i++) {
192 const auto& bc_config = bc_configs[i];
193 std::string bc_name = bc_config["bc_name"];
194 std::string bc_type = bc_config["bc_type"];
195 bc_type_map.insert({bc_name, bc_type});
196 }
197
198 // Create external coupling blocks
199 if (config.contains("external_solver_coupling_blocks")) {
200 DEBUG_MSG("Create external coupling blocks");
201 const auto& coupling_configs = config["external_solver_coupling_blocks"];
202 for (const auto& coupling_config : coupling_configs) {
203 std::string coupling_type = coupling_config["type"];
204 std::string coupling_name = coupling_config["name"];
205 std::string coupling_loc = coupling_config["location"];
206 bool periodic = coupling_config.value("periodic", true);
207 const auto& coupling_values = coupling_config["values"];
208
209 // Create coupling block
210 auto t_coupling = get_double_array(coupling_values, "t", {0.0});
211
212 if (coupling_type == "FLOW") {
213 auto Q_coupling = get_double_array(coupling_values, "Q");
214 model.add_block(MODEL::BlockType::FLOWBC,
215 {model.add_parameter(t_coupling, Q_coupling, periodic)},
216 coupling_name);
217 } else if (coupling_type == "PRESSURE") {
218 auto P_coupling = get_double_array(coupling_values, "P");
219 model.add_block(MODEL::BlockType::PRESSUREBC,
220 {model.add_parameter(t_coupling, P_coupling, periodic)},
221 coupling_name);
222 } else {
223 throw std::runtime_error(
224 "Error. Flowsolver coupling block types should be FLOW or "
225 "PRESSURE.");
226 }
227 DEBUG_MSG("Created coupling block " << coupling_name);
228
229 // Determine the type of connected block
230 std::string connected_block = coupling_config["connected_block"];
231 std::string connected_type;
232 int found_block = 0;
233 if (connected_block == "ClosedLoopHeartAndPulmonary") {
234 connected_type = "ClosedLoopHeartAndPulmonary";
235 found_block = 1;
236 } else {
237 try {
238 connected_type = bc_type_map.at(connected_block);
239 found_block = 1;
240 } catch (...) {
241 }
242 if (found_block == 0) {
243 // Search for connected_block in the list of vessel names
244 for (auto const vessel : vessel_id_map) {
245 if (connected_block == vessel.second) {
246 connected_type = "BloodVessel";
247 found_block = 1;
248 break;
249 }
250 }
251 }
252 if (found_block == 0) {
253 std::cout << "Error! Could not connected type for block: "
254 << connected_block << std::endl;
255 throw std::runtime_error("Terminating.");
256 }
257 } // connected_block != "ClosedLoopHeartAndPulmonary"
258 // Create connections
259 if (coupling_loc == "inlet") {
260 std::vector<std::string> possible_types = {
261 "RESISTANCE", "RCR", "ClosedLoopRCR",
262 "SimplifiedRCR", "CORONARY", "ClosedLoopCoronary",
263 "BloodVessel"};
264 if (std::find(std::begin(possible_types), std::end(possible_types),
265 connected_type) == std::end(possible_types)) {
266 throw std::runtime_error(
267 "Error: The specified connection type for inlet "
268 "external_coupling_block is invalid.");
269 }
270 connections.push_back({coupling_name, connected_block});
271 DEBUG_MSG("Created coupling block connection: " << coupling_name << "->"
272 << connected_block);
273 } else if (coupling_loc == "outlet") {
274 std::vector<std::string> possible_types = {
275 "ClosedLoopRCR", "ClosedLoopHeartAndPulmonary", "BloodVessel"};
276 if (std::find(std::begin(possible_types), std::end(possible_types),
277 connected_type) == std::end(possible_types)) {
278 throw std::runtime_error(
279 "Error: The specified connection type for outlet "
280 "external_coupling_block is invalid.");
281 }
282 // Add connection only for closedLoopRCR and BloodVessel. Connection to
283 // ClosedLoopHeartAndPulmonary will be handled in
284 // ClosedLoopHeartAndPulmonary creation.
285 if ((connected_type == "ClosedLoopRCR") ||
286 (connected_type == "BloodVessel")) {
287 connections.push_back({connected_block, coupling_name});
288 DEBUG_MSG("Created coupling block connection: "
289 << connected_block << "-> " << coupling_name);
290 } // connected_type == "ClosedLoopRCR"
291 } // coupling_loc
292 } // for (size_t i = 0; i < coupling_configs.length(); i++)
293 }
294
295 // Create boundary conditions
296 std::vector<std::string> closed_loop_bcs;
297 DEBUG_MSG("Create boundary conditions");
298 for (size_t i = 0; i < bc_configs.size(); i++) {
299 const auto& bc_config = bc_configs[i];
300 std::string bc_type = bc_config["bc_type"];
301 std::string bc_name = bc_config["bc_name"];
302 const auto& bc_values = bc_config["bc_values"];
303
304 auto t = get_double_array(bc_values, "t", {0.0});
305 if (bc_type == "RCR") {
306 model.add_block(MODEL::BlockType::WINDKESSELBC,
307 {
308 model.add_parameter(bc_values["Rp"]),
309 model.add_parameter(bc_values["C"]),
310 model.add_parameter(bc_values["Rd"]),
311 model.add_parameter(bc_values["Pd"]),
312 },
313 bc_name);
314 } else if (bc_type == "ClosedLoopRCR") {
315 model.add_block(MODEL::BlockType::CLOSEDLOOPRCRBC,
316 {model.add_parameter(bc_values["Rp"]),
317 model.add_parameter(bc_values["C"]),
318 model.add_parameter(bc_values["Rd"])},
319 bc_name);
320 if (bc_values["closed_loop_outlet"] == true) {
321 closed_loop_bcs.push_back(bc_name);
322 }
323
324 } else if (bc_type == "FLOW") {
325 model.add_block(
326 MODEL::BlockType::FLOWBC,
327 {model.add_parameter(t, get_double_array(bc_values, "Q"))}, bc_name);
328
329 } else if (bc_type == "RESISTANCE") {
330 model.add_block(
331 MODEL::BlockType::RESISTANCEBC,
332 {model.add_parameter(t, get_double_array(bc_values, "R")),
333 model.add_parameter(t, get_double_array(bc_values, "Pd"))},
334 bc_name);
335
336 } else if (bc_type == "PRESSURE") {
337 model.add_block(
338 MODEL::BlockType::PRESSUREBC,
339 {model.add_parameter(t, get_double_array(bc_values, "P"))}, bc_name);
340
341 } else if (bc_type == "CORONARY") {
342 model.add_block(
343 MODEL::BlockType::OPENLOOPCORONARYBC,
344 {model.add_parameter(t, get_double_array(bc_values, "Ra1")),
345 model.add_parameter(t, get_double_array(bc_values, "Ra2")),
346 model.add_parameter(t, get_double_array(bc_values, "Rv1")),
347 model.add_parameter(t, get_double_array(bc_values, "Ca")),
348 model.add_parameter(t, get_double_array(bc_values, "Cc")),
349 model.add_parameter(t, get_double_array(bc_values, "Pim")),
350 model.add_parameter(t, get_double_array(bc_values, "P_v"))},
351 bc_name);
352
353 } else if (bc_type == "ClosedLoopCoronary") {
354 std::string side = bc_values["side"];
355 MODEL::BlockType block_type;
356 if (side == "left") {
357 block_type = MODEL::BlockType::CLOSEDLOOPCORONARYLEFTBC;
358 } else if (side == "right") {
359 block_type = MODEL::BlockType::CLOSEDLOOPCORONARYRIGHTBC;
360 } else {
361 throw std::runtime_error("Invalid side for ClosedLoopCoronary");
362 }
363 model.add_block(block_type,
364 {model.add_parameter(bc_values["Ra"]),
365 model.add_parameter(bc_values["Ram"]),
366 model.add_parameter(bc_values["Rv"]),
367 model.add_parameter(bc_values["Ca"]),
368 model.add_parameter(bc_values["Cim"])},
369 bc_name);
370 closed_loop_bcs.push_back(bc_name);
371
372 } else {
373 throw std::invalid_argument("Unknown boundary condition type");
374 }
375 DEBUG_MSG("Created boundary condition " << bc_name);
376 }
377
378 // Create junctions
379 for (const auto& junction_config : config["junctions"]) {
380 std::string j_type = junction_config["junction_type"];
381 std::string junction_name = junction_config["junction_name"];
382 if ((j_type == "NORMAL_JUNCTION") || (j_type == "internal_junction")) {
383 model.add_block(MODEL::BlockType::JUNCTION, {}, junction_name);
384 } else if (j_type == "resistive_junction") {
385 const auto& junction_values = junction_config["junction_values"];
386 std::vector<int> param_ids;
387 for (T value : junction_values["R"]) {
388 param_ids.push_back(model.add_parameter(value));
389 }
390 model.add_block(MODEL::BlockType::RESISTIVEJUNCTION, param_ids,
391 junction_name);
392 } else if (j_type == "BloodVesselJunction") {
393 const auto& junction_values = junction_config["junction_values"];
394 std::vector<int> param_ids;
395 for (T value : junction_values["R_poiseuille"]) {
396 param_ids.push_back(model.add_parameter(value));
397 }
398 for (T value : junction_values["L"]) {
399 param_ids.push_back(model.add_parameter(value));
400 }
401 for (T value : junction_values["stenosis_coefficient"]) {
402 param_ids.push_back(model.add_parameter(value));
403 }
404 model.add_block(MODEL::BlockType::BLOODVESSELJUNCTION, param_ids,
405 junction_name);
406 } else {
407 throw std::invalid_argument("Unknown junction type");
408 }
409 // Check for connections to inlet and outlet vessels and append to
410 // connections list
411 for (int vessel_id : junction_config["inlet_vessels"]) {
412 connections.push_back({vessel_id_map[vessel_id], junction_name});
413 }
414 for (int vessel_id : junction_config["outlet_vessels"]) {
415 connections.push_back({junction_name, vessel_id_map[vessel_id]});
416 }
417 DEBUG_MSG("Created junction " << junction_name);
418 }
419
420 // Create closed-loop blocks
421 bool heartpulmonary_block_present =
422 false; ///< Flag to check if heart block is present (requires different
423 ///< handling)
424 if (config.contains("closed_loop_blocks")) {
425 for (const auto& closed_loop_config : config["closed_loop_blocks"]) {
426 std::string closed_loop_type = closed_loop_config["closed_loop_type"];
427 if (closed_loop_type == "ClosedLoopHeartAndPulmonary") {
428 if (heartpulmonary_block_present == false) {
429 heartpulmonary_block_present = true;
430 std::string heartpulmonary_name = "CLH";
431 T cycle_period = closed_loop_config["cardiac_cycle_period"];
432 if ((model.cardiac_cycle_period > 0.0) &&
433 (cycle_period != model.cardiac_cycle_period)) {
434 throw std::runtime_error(
435 "Inconsistent cardiac cycle period defined in "
436 "ClosedLoopHeartAndPulmonary.");
437 } else {
438 model.cardiac_cycle_period = cycle_period;
439 }
440 const auto& heart_params = closed_loop_config["parameters"];
441 // Convert to std::map to keep model blocks independent of simdjson
442 model.add_block(MODEL::BlockType::CLOSEDLOOPHEARTPULMONARY,
443 {model.add_parameter(heart_params["Tsa"]),
444 model.add_parameter(heart_params["tpwave"]),
445 model.add_parameter(heart_params["Erv_s"]),
446 model.add_parameter(heart_params["Elv_s"]),
447 model.add_parameter(heart_params["iml"]),
448 model.add_parameter(heart_params["imr"]),
449 model.add_parameter(heart_params["Lra_v"]),
450 model.add_parameter(heart_params["Rra_v"]),
451 model.add_parameter(heart_params["Lrv_a"]),
452 model.add_parameter(heart_params["Rrv_a"]),
453 model.add_parameter(heart_params["Lla_v"]),
454 model.add_parameter(heart_params["Rla_v"]),
455 model.add_parameter(heart_params["Llv_a"]),
456 model.add_parameter(heart_params["Rlv_ao"]),
457 model.add_parameter(heart_params["Vrv_u"]),
458 model.add_parameter(heart_params["Vlv_u"]),
459 model.add_parameter(heart_params["Rpd"]),
460 model.add_parameter(heart_params["Cp"]),
461 model.add_parameter(heart_params["Cpa"]),
462 model.add_parameter(heart_params["Kxp_ra"]),
463 model.add_parameter(heart_params["Kxv_ra"]),
464 model.add_parameter(heart_params["Kxp_la"]),
465 model.add_parameter(heart_params["Kxv_la"]),
466 model.add_parameter(heart_params["Emax_ra"]),
467 model.add_parameter(heart_params["Emax_la"]),
468 model.add_parameter(heart_params["Vaso_ra"]),
469 model.add_parameter(heart_params["Vaso_la"])},
470 heartpulmonary_name);
471
472 // Junction at inlet to heart
473 std::string heart_inlet_junction_name = "J_heart_inlet";
474 connections.push_back(
475 {heart_inlet_junction_name, heartpulmonary_name});
476 model.add_block(MODEL::BlockType::JUNCTION, {},
477 heart_inlet_junction_name);
478 for (auto heart_inlet_elem : closed_loop_bcs) {
479 connections.push_back(
480 {heart_inlet_elem, heart_inlet_junction_name});
481 }
482
483 // Junction at outlet from heart
484 std::string heart_outlet_junction_name = "J_heart_outlet";
485 connections.push_back(
486 {heartpulmonary_name, heart_outlet_junction_name});
487 model.add_block(MODEL::BlockType::JUNCTION, {},
488 heart_outlet_junction_name);
489 for (auto& outlet_block : closed_loop_config["outlet_blocks"]) {
490 connections.push_back({heart_outlet_junction_name, outlet_block});
491 }
492 } else {
493 throw std::runtime_error(
494 "Error. Only one ClosedLoopHeartAndPulmonary can be included.");
495 }
496 }
497 }
498 }
499
500 // Create Connections
501 for (auto& connection : connections) {
502 auto ele1 = model.get_block(std::get<0>(connection));
503 auto ele2 = model.get_block(std::get<1>(connection));
504 model.add_node({ele1}, {ele2}, ele1->get_name() + ":" + ele2->get_name());
505 }
506 // Finalize model
507 model.finalize();
508}

510/**
511 * @brief Load initial conditions from a configuration
512 *
513 * @tparam T Scalar type (e.g. `float`, `double`)
514 * @param config The json configuration
515 * @param model The model
516 * @return ALGEBRA::State<T> Initial configuration for the model
517 */
518template <typename T>
519ALGEBRA::State<T> load_initial_condition(const nlohmann::json& config,
520 MODEL::Model<T>& model) {
521 // Read initial condition
522 auto initial_state = ALGEBRA::State<T>::Zero(model.dofhandler.size());
523
524 if (config.contains("initial_condition")) {
525 const auto& initial_condition = config["initial_condition"];
526 // Check for pressure_all or flow_all condition.
527 // This will initialize all pressure:* and flow:* variables.
528 T init_p, init_q;
529 bool init_p_flag = initial_condition.contains("pressure_all");
530 bool init_q_flag = initial_condition.contains("flow_all");
531 if (init_p_flag) {
532 init_p = initial_condition["pressure_all"];
533 }
534 if (init_q_flag) {
535 init_q = initial_condition["flow_all"];
536 }
537
538 // Loop through variables and check for initial conditions.
539 for (size_t i = 0; i < model.dofhandler.size(); i++) {
540 std::string var_name = model.dofhandler.variables[i];
541 double default_val = 0.0;
542 // If initial condition is not specified for this variable,
543 // check if pressure_all/flow_all are applicable
544 if (!initial_condition.contains(var_name)) {
545 if ((init_p_flag == true) && ((var_name.substr(0, 9) == "pressure:") ||
546 (var_name.substr(0, 4) == "P_c:"))) {
547 default_val = init_p;
548 DEBUG_MSG("pressure_all initial condition for " << var_name);
549 } else if ((init_q_flag == true) &&
550 (var_name.substr(0, 5) == "flow:")) {
551 default_val = init_q;
552 DEBUG_MSG("flow_all initial condition for " << var_name);
553 } else {
554 DEBUG_MSG("No initial condition found for "
555 << var_name << ". Using default value = 0.");
556 }
557 }
558 initial_state.y[i] = initial_condition.value(var_name, default_val);
559 }
560 }
561 if (config.contains("initial_condition_d")) {
562 DEBUG_MSG("Reading initial condition derivative");
563 const auto& initial_condition_d = config["initial_condition_d"];
564 // Loop through variables and check for initial conditions.
565 for (size_t i = 0; i < model.dofhandler.size(); i++) {
566 std::string var_name = model.dofhandler.variables[i];
567 if (!initial_condition_d.contains(var_name)) {
568 DEBUG_MSG("No initial condition derivative found for " << var_name);
569 }
570 initial_state.ydot[i] = initial_condition_d.value(var_name, 0.0);
571 }
572 }
573 return initial_state;
574}
