126template <typename T>
127SparseSystem<T>::SparseSystem() {}

129template <typename T>
130SparseSystem<T>::SparseSystem(unsigned int n) {
131 F = Eigen::SparseMatrix<T>(n, n);
132 E = Eigen::SparseMatrix<T>(n, n);
133 D = Eigen::SparseMatrix<T>(n, n);
134 C = Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n);
135
136 jacobian = Eigen::SparseMatrix<T>(n, n);
137 residual = Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n);
138 dy = Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(n);
139}

146 // Cannot be in destructor because dynamically allocated pointers will be lost
147 // when objects are assigned from temporary objects.
148 delete solver;

151template <typename T>
152void SparseSystem<T>::reserve(MODEL::Model<T> *model) {
153 auto num_triplets = model->get_num_triplets();
154 F.reserve(num_triplets["F"]);
155 E.reserve(num_triplets["E"]);
156 D.reserve(num_triplets["D"]);
157 model->update_constant(*this);
158 model->update_time(*this, 0.0);
159 Eigen::Matrix<T, Eigen::Dynamic, 1> dummy_y =
160 Eigen::Matrix<T, Eigen::Dynamic, 1>::Ones(residual.size());
161 Eigen::Matrix<T, Eigen::Dynamic, 1> dummy_dy =
162 Eigen::Matrix<T, Eigen::Dynamic, 1>::Ones(residual.size());
163 model->update_solution(*this, dummy_y, dummy_dy);
164 F.makeCompressed();
165 E.makeCompressed();
166 D.makeCompressed();
167 jacobian.reserve(num_triplets["F"] + num_triplets["E"]); // Just an estimate
168 update_jacobian(1.0); // Update it once to have sparsity pattern
169 jacobian.makeCompressed();
170 solver->analyzePattern(jacobian); // Let solver analyze pattern
171}

173template <typename T>
174void SparseSystem<T>::update_residual(
175 Eigen::Matrix<T, Eigen::Dynamic, 1> &y,
176 Eigen::Matrix<T, Eigen::Dynamic, 1> &ydot) {
177 residual.setZero();
178 residual -= C;
179 residual.noalias() -= E * ydot;
180 residual.noalias() -= F * y;
181}
182
183template <typename T>
184void SparseSystem<T>::update_jacobian(T e_coeff) {
185 jacobian.setZero();
186 jacobian += F + D + E * e_coeff;
187}
188
189template <typename T>
190void SparseSystem<T>::solve() {
191 solver->factorize(jacobian);
192 dy.setZero();
193 dy += solver->solve(residual);
194}
