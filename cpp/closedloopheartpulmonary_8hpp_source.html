45/**
46 * @brief Heart and pulmonary circulation model
47 *
48 * Models the mechanics of the 4 heart chambers and pulmonary circulation
49 *
50 * Reference for equations and model structure: Sankaran, S., Moghadam, M. E.,
51 * Kahn, A. M., Tseng, E. E., Guccione, J. M., & Marsden, A. L. (2012).
52 * Patient-specific multiscale modeling of blood flow for coronary artery bypass
53 * graft surgery. Annals of Biomedical Engineering, 40(10), 2228–2242.
54 * https://doi.org/10.1007/s10439-012-0579-3
55 *
56 * TODO: Equations and circuit diagram
57 *
58 * ### Parameters
59 *
60 * Parameter sequence for constructing this block
61 *
62 * * `0` Atrial systole time fraction
63 * * `1` Time for P-wave
64 * * `2` Scaling for right ventricle elastance
65 * * `3` Scaling for left ventricle elastance
66 * * `4` Scaling for intramyocardial pressure (left coronaries)
67 * * `5` Scaling for intramyocardial pressure (right coronaries)
68 * * `6` Right atrium inductance
69 * * `7` Right atrium outflow resistance
70 * * `8` Right ventricle inductance
71 * * `9` Right ventricle outflow resistance
72 * * `10` Left atrium inductance
73 * * `11` Left atrium outflow resistance
74 * * `12` Left ventricle inductance
75 * * `13` Left ventricle outflow resistance
76 * * `14` Right ventricle unstressed volume
77 * * `15` Left ventricle unstressed volume
78 * * `16` Pulmonary resistance
79 * * `17` Pulmonary capacitance
80 * * `18` Aortic capacitance
81 * * `19` Right atrium pressure scaling
82 * * `20` Right atrium volume scaling
83 * * `21` Left atrium pressure scaling
84 * * `22` Left atrium volume scaling
85 * * `23` Right atrium elastance
86 * * `24` Left atrium elastance
87 * * `25` Right atrium resting volume
88 * * `26` Left atrium resting volume
89 *
90 * @tparam T Scalar type (e.g. `float`, `double`)
91 */
92template <typename T>
93class ClosedLoopHeartPulmonary : public Block<T> {
94 public:
95 // Inherit constructors
96 using Block<T>::Block;

102 enum ParamId {
103 TSA = 0, ///< Fractions of cardiac cycle (not sure)
104 TPWAVE = 1, ///< Fraction of cardiac cycle (P-wave)
105 ERV_S = 2, ///< Scaling for right ventricle elastance
106 ELV_S = 3, ///< Scaling for left ventricle elastance
107 IML = 4, ///< Scaling for intramyocardial pressure (left coronaries)
108 IMR = 5, ///< Scaling for intramyocardial pressure (right coronaries)
109 LRA_V = 6, ///< Right atrium inductance
110 RRA_V = 7, ///< Right atrium outflow resistance
111 LRV_A = 8, ///< Right ventricle inductance
112 RRV_A = 9, ///< Right ventricle outflow resistance
113 LLA_V = 10, ///< Left atrium inductance
114 RLA_V = 11, ///< Left atrium outflow resistance
115 LLV_A = 12, ///< Left ventricle inductance
116 RLV_AO = 13, ///< Left ventricle outflow resistance
117 VRV_U = 14, ///< Right ventricle unstressed volume
118 VLV_U = 15, ///< Left ventricle unstressed volume
119 RPD = 16, ///< Pulmonary resistance
120 CP = 17, ///< Pulmonary capacitance
121 CPA = 18, ///< Aortic capacitance
122 KXP_RA = 19, ///< Right atrium pressure-volume relationship (?)
123 KXV_RA = 20, ///< Right atrium pressure-volume relationship (?)
124 KXP_LA = 21, ///< Left atrium pressure-volume relationship (?)
125 KXV_LA = 22, ///< Left atrium pressure-volume relationship (?)
126 EMAX_RA = 23, ///< Right atrium elastance (?)
127 EMAX_LA = 24, ///< Left atrium elastance (?)
128 VASO_RA = 25, ///< Right atrium rest volume (?)

241template <typename T>
242void ClosedLoopHeartPulmonary<T>::update_constant(
243 ALGEBRA::SparseSystem<T> &system, std::vector<T> &parameters) {
244 // DOF 2, Eq 1: Aortic pressure
245 system.E.coeffRef(this->global_eqn_ids[1], this->global_var_ids[2]) =
246 parameters[this->global_param_ids[ParamId::CPA]];
247 // DOF 4, Eq 2: Right atrium volume
248 system.E.coeffRef(this->global_eqn_ids[2], this->global_var_ids[4]) = 1.0;
249 // DOF 5, Eq 3: Right atrium outflow
250 system.E.coeffRef(this->global_eqn_ids[3], this->global_var_ids[5]) =
251 parameters[this->global_param_ids[ParamId::LRA_V]];
252 // DOF 7, Eq 5: Right ventricle volume
253 system.E.coeffRef(this->global_eqn_ids[5], this->global_var_ids[7]) = 1.0;
254 // DOF 8, Eq 6: Right ventricle outflow
255 system.E.coeffRef(this->global_eqn_ids[6], this->global_var_ids[8]) =
256 parameters[this->global_param_ids[ParamId::LRV_A]];
257 // DOF 9, Eq 7: Pulmonary pressure
258 system.E.coeffRef(this->global_eqn_ids[7], this->global_var_ids[9]) =
259 parameters[this->global_param_ids[ParamId::CP]];
260 // DOF 11, Eq 9: Left atrium volume
261 system.E.coeffRef(this->global_eqn_ids[9], this->global_var_ids[11]) = 1.0;
262 // DOF 12, Eq 10: Left atrium outflow
263 system.E.coeffRef(this->global_eqn_ids[10], this->global_var_ids[12]) =
264 parameters[this->global_param_ids[ParamId::LLA_V]];
265 // DOF 14, Eq 12: Left ventricle volume
266 system.E.coeffRef(this->global_eqn_ids[12], this->global_var_ids[14]) = 1.0;
267 // DOF 15, Eq 13: Left ventricle outflow
268 system.E.coeffRef(this->global_eqn_ids[13], this->global_var_ids[15]) =
269 parameters[this->global_param_ids[ParamId::LLV_A]];
270}

278template <typename T>
279void ClosedLoopHeartPulmonary<T>::update_solution(
280 ALGEBRA::SparseSystem<T> &system, std::vector<T> &parameters,
281 Eigen::Matrix<T, Eigen::Dynamic, 1> &y,
282 Eigen::Matrix<T, Eigen::Dynamic, 1> &dy) {
283 this->get_psi_ra_la(parameters, y);
284 this->get_valve_positions(y);
285
286 // F and C matrices depend on time and solution
287 // Specifying all terms here, including constant terms (which can instead be
288 // specified in update_constant) for readability (Doesn't seem to make a
289 // difference to compute time) DOF IDs are arranged as inflow
290 // [P_in,Q_in,P_out,Q_out,internal variables...]
291
292 // DOF 0, Eq 0: Right atrium pressure
293 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[0]) = 1.0;
294 system.F.coeffRef(this->global_eqn_ids[0], this->global_var_ids[4]) =
295 -this->AA * parameters[this->global_param_ids[ParamId::EMAX_RA]];
296 system.C(this->global_eqn_ids[0]) =
297 this->AA * parameters[this->global_param_ids[ParamId::EMAX_RA]] *
298 parameters[this->global_param_ids[ParamId::VASO_RA]] +
299 psi_ra * (this->AA - 1.0);
300 system.D.coeffRef(this->global_eqn_ids[0], this->global_var_ids[4]) =
301 psi_ra_derivative * (this->AA - 1.0);
302
303 // DOF 1: Flow into right atrium (no equation)
304
305 // DOF 2, Eq 1: Aortic pressure
306 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[15]) =
307 -valves[15];
308 system.F.coeffRef(this->global_eqn_ids[1], this->global_var_ids[3]) = 1.0;
309
310 // DOF 3: Flow into aorta (no equation)
311
312 // DOF 4, Eq 2: Right atrium volume
313 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[5]) =
314 1.0 * valves[5];
315 system.F.coeffRef(this->global_eqn_ids[2], this->global_var_ids[1]) = -1.0;
316
317 // DOF 5, Eq 3: Right atrium outflow
318 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[5]) =
319 parameters[this->global_param_ids[ParamId::RRA_V]] * valves[5];
320 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[0]) = -1.0;
321 system.F.coeffRef(this->global_eqn_ids[3], this->global_var_ids[6]) = 1.0;
322
323 // DOF 6, Eq 4: Right ventricle pressure
324 system.F.coeffRef(this->global_eqn_ids[4], this->global_var_ids[6]) = 1.0;
325 system.F.coeffRef(this->global_eqn_ids[4], this->global_var_ids[7]) =
326 -this->Erv;
327 system.C(this->global_eqn_ids[4]) =
328 this->Erv * parameters[this->global_param_ids[ParamId::VRV_U]];
329
330 // DOF 7, Eq 5: Right ventricle volume
331 system.F.coeffRef(this->global_eqn_ids[5], this->global_var_ids[5]) =
332 -1.0 * valves[5];
333 system.F.coeffRef(this->global_eqn_ids[5], this->global_var_ids[8]) =
334 1.0 * valves[8];
335
336 // DOF 8, Eq 6: Right ventricle outflow
337 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[6]) = -1.0;
338 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[9]) = 1.0;
339 system.F.coeffRef(this->global_eqn_ids[6], this->global_var_ids[8]) =
340 parameters[this->global_param_ids[ParamId::RRV_A]] * valves[8];
341
342 // DOF 9, Eq 7: Pulmonary pressure
343 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[8]) =
344 -valves[8];
345 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[9]) =
346 1.0 / parameters[this->global_param_ids[ParamId::RPD]];
347 system.F.coeffRef(this->global_eqn_ids[7], this->global_var_ids[10]) =
348 -1.0 / parameters[this->global_param_ids[ParamId::RPD]];
349
350 // DOF 10, Eq 8: Left atrium pressure
351 system.F.coeffRef(this->global_eqn_ids[8], this->global_var_ids[10]) = 1.0;
352 system.F.coeffRef(this->global_eqn_ids[8], this->global_var_ids[11]) =
353 -this->AA * parameters[this->global_param_ids[ParamId::EMAX_LA]];
354 system.C(this->global_eqn_ids[8]) =
355 this->AA * parameters[this->global_param_ids[ParamId::EMAX_LA]] *
356 parameters[this->global_param_ids[ParamId::VASO_LA]] +
357 psi_la * (this->AA - 1.0);
358 system.D.coeffRef(this->global_eqn_ids[8], this->global_var_ids[11]) =
359 psi_la_derivative * (this->AA - 1.0);
360
361 // DOF 11, Eq 9: Left atrium volume
362 system.F.coeffRef(this->global_eqn_ids[9], this->global_var_ids[8]) =
363 -1.0 * valves[8];
364 system.F.coeffRef(this->global_eqn_ids[9], this->global_var_ids[12]) =
365 1.0 * valves[12];
366
367 // DOF 12, Eq 10: Left atrium outflow
368 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[10]) = -1.0;
369 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[13]) = 1.0;
370 system.F.coeffRef(this->global_eqn_ids[10], this->global_var_ids[12]) =
371 parameters[this->global_param_ids[ParamId::RLA_V]] * valves[12];
372
373 // DOF 13, Eq 11: Left ventricle pressure
374 system.F.coeffRef(this->global_eqn_ids[11], this->global_var_ids[13]) = 1.0;
375 system.F.coeffRef(this->global_eqn_ids[11], this->global_var_ids[14]) =
376 -this->Elv;
377 system.C(this->global_eqn_ids[11]) =
378 this->Elv * parameters[this->global_param_ids[ParamId::VLV_U]];
379
380 // DOF 14, Eq 12: Left ventricle volume
381 system.F.coeffRef(this->global_eqn_ids[12], this->global_var_ids[12]) =
382 -1.0 * valves[12];
383 system.F.coeffRef(this->global_eqn_ids[12], this->global_var_ids[15]) =
384 1.0 * valves[15];
385
386 // DOF 15, Eq 13: Left ventricle outflow
387 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[13]) = -1.0;
388 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[2]) = 1.0;
389 system.F.coeffRef(this->global_eqn_ids[13], this->global_var_ids[15]) =
390 parameters[this->global_param_ids[ParamId::RLV_AO]] * valves[15];
391}

393template <typename T>
394void ClosedLoopHeartPulmonary<T>::get_activation_and_elastance_functions(
395 std::vector<T> &parameters) {
396 T T_cardiac = this->model->cardiac_cycle_period;
397 T Tsa = T_cardiac * parameters[this->global_param_ids[ParamId::TSA]];
398 T tpwave = T_cardiac / parameters[this->global_param_ids[ParamId::TPWAVE]];
399 T t_in_cycle = fmod(this->model->time, T_cardiac);
400
401 // Activation function
402 AA = 0.0;
403 if (t_in_cycle <= tpwave) {
404 AA = (0.5) * (1.0 - cos(2.0 * PI * (t_in_cycle - tpwave + Tsa) / Tsa));
405 } else if ((t_in_cycle >= (T_cardiac - Tsa) + tpwave) and
406 (t_in_cycle < T_cardiac)) {
407 AA =
408 (0.5) *
409 (1.0 - cos(2.0 * PI * (t_in_cycle - tpwave - (T_cardiac - Tsa)) / Tsa));
410 } else {
411 AA = 0.0;
412 }
413
414 // Elastance modes (copied from J. Tran's tuning framework)
415 const int num_elast_modes = 25;
416 T Ft_elastance[num_elast_modes][2] = {
417 {0.283748803, 0.000000000}, {0.031830626, -0.374299825},
418 {-0.209472400, -0.018127770}, {0.020520047, 0.073971113},
419 {0.008316883, -0.047249597}, {-0.041677660, 0.003212163},
420 {0.000867323, 0.019441411}, {-0.001675379, -0.005565534},
421 {-0.011252277, 0.003401432}, {-0.000414677, 0.008376795},
422 {0.000253749, -0.000071880}, {-0.002584966, 0.001566861},
423 {0.000584752, 0.003143555}, {0.000028502, -0.000024787},
424 {0.000022961, -0.000007476}, {0.000018735, -0.000001281},
425 {0.000015573, 0.000001781}, {0.000013133, 0.000003494},
426 {0.000011199, 0.000004507}, {0.000009634, 0.000005117},
427 {0.000008343, 0.000005481}, {0.000007265, 0.000005687},
428 {0.000006354, 0.000005789}, {0.000005575, 0.000005821},
429 {0.000004903, 0.000005805}};
430
431 // RV and LV elastance
432 T Elv_i = 0.0;
433 for (auto i = 0; i < num_elast_modes; i++)
434 Elv_i = Elv_i +
435 (Ft_elastance[i][0]) * cos(2.0 * PI * i * t_in_cycle / T_cardiac) -
436 (Ft_elastance[i][1]) * sin(2.0 * PI * i * t_in_cycle / T_cardiac);
437
438 Elv = Elv_i * parameters[this->global_param_ids[ParamId::ELV_S]];
439 Erv = Elv_i * parameters[this->global_param_ids[ParamId::ERV_S]];
440}

442template <typename T>
443void ClosedLoopHeartPulmonary<T>::get_psi_ra_la(
444 std::vector<T> &parameters, Eigen::Matrix<T, Eigen::Dynamic, 1> &y) {
445 auto RA_volume = y[this->global_var_ids[4]];
446 auto LA_volume = y[this->global_var_ids[11]];
447 psi_ra =
448 parameters[this->global_param_ids[ParamId::KXP_RA]] *
449 (exp((RA_volume - parameters[this->global_param_ids[ParamId::VASO_RA]]) *
450 parameters[this->global_param_ids[ParamId::KXV_RA]]) -
451 1.0);
452 psi_la =
453 parameters[this->global_param_ids[ParamId::KXP_LA]] *
454 (exp((LA_volume - parameters[this->global_param_ids[ParamId::VASO_LA]]) *
455 parameters[this->global_param_ids[ParamId::KXV_LA]]) -
456 1.0);
457
458 psi_ra_derivative =
459 parameters[this->global_param_ids[ParamId::KXP_RA]] *
460 exp((RA_volume - parameters[this->global_param_ids[ParamId::VASO_RA]]) *
461 parameters[this->global_param_ids[ParamId::KXV_RA]]) *
462 parameters[this->global_param_ids[ParamId::KXV_RA]];
463 psi_la_derivative =
464 parameters[this->global_param_ids[ParamId::KXP_LA]] *
465 exp((LA_volume - parameters[this->global_param_ids[ParamId::VASO_LA]]) *
466 parameters[this->global_param_ids[ParamId::KXV_LA]]) *
467 parameters[this->global_param_ids[ParamId::KXV_LA]];
468}

470template <typename T>
471void ClosedLoopHeartPulmonary<T>::get_valve_positions(
472 Eigen::Matrix<T, Eigen::Dynamic, 1> &y) {
473 std::fill(valves, valves + 16, 1.0);
474
475 // RA to RV
476 auto pressure_ra = y[this->global_var_ids[0]];
477 auto pressure_rv = y[this->global_var_ids[6]];
478 auto outflow_ra = y[this->global_var_ids[5]];
479 if ((pressure_ra <= pressure_rv) and (outflow_ra <= 0.0)) {
480 valves[5] = 0.0;
481 y[this->global_var_ids[5]] = 0.0;
482 }
483
484 // RV to pulmonary
485 auto pressure_pulmonary = y[this->global_var_ids[9]];
486 auto outflow_rv = y[this->global_var_ids[8]];
487 if ((pressure_rv <= pressure_pulmonary) and (outflow_rv <= 0.0)) {
488 valves[8] = 0.0;
489 y[this->global_var_ids[8]] = 0.0;
490 }
491
492 // LA to LV
493 auto pressure_la = y[this->global_var_ids[10]];
494 auto pressure_lv = y[this->global_var_ids[13]];
495 auto outflow_la = y[this->global_var_ids[12]];
496 if ((pressure_la <= pressure_lv) and (outflow_la <= 0.0)) {
497 valves[12] = 0.0;
498 y[this->global_var_ids[12]] = 0.0;
499 }
500
501 // LV to aorta
502 auto pressure_aorta = y[this->global_var_ids[2]];
503 auto outflow_lv = y[this->global_var_ids[15]];
504 if ((pressure_lv <= pressure_aorta) and (outflow_lv <= 0.0)) {
505 valves[15] = 0.0;
506 y[this->global_var_ids[15]] = 0.0;
507 }
508}
