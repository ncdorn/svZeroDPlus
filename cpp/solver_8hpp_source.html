130template <typename T>
131Solver<T>::Solver(const nlohmann::json& config) {
132 DEBUG_MSG("Read simulation parameters");
133 simparams = IO::load_simulation_params<T>(config);
134 DEBUG_MSG("Load model");
135 model = MODEL::Model<T>();
136 IO::load_simulation_model<T>(config, model);
137 DEBUG_MSG("Load initial condition");
138 initial_state = IO::load_initial_condition<T>(config, model);
139
140 DEBUG_MSG("Cardiac cycle period " << model.cardiac_cycle_period);
141
142 // Calculate time step size
143 if (!simparams.sim_coupled) {
144 simparams.sim_time_step_size =
145 model.cardiac_cycle_period / (T(simparams.sim_pts_per_cycle) - 1.0);
146 } else {
147 simparams.sim_time_step_size = simparams.sim_external_step_size /
148 (T(simparams.sim_num_time_steps) - 1.0);
149 }
150 sanity_checks();
151}

156template <typename T>
157void Solver<T>::run() {
158 auto state = initial_state;
159
160 // Create steady initial
161 if (simparams.sim_steady_initial) {
162 DEBUG_MSG("Calculate steady initial condition");
163 T time_step_size_steady = model.cardiac_cycle_period / 10.0;
164 model.to_steady();
165 ALGEBRA::Integrator<T> integrator_steady(&model, time_step_size_steady, 0.1,
166 simparams.sim_abs_tol,
167 simparams.sim_nliter);
168 for (int i = 0; i < 31; i++) {
169 state = integrator_steady.step(state, time_step_size_steady * T(i));
170 }
171 model.to_unsteady();
172 }
173
174 // Set-up integrator
175 DEBUG_MSG("Setup time integration");
176 ALGEBRA::Integrator<T> integrator(&model, simparams.sim_time_step_size, 0.1,
177 simparams.sim_abs_tol,
178 simparams.sim_nliter);
179
180 // Initialize loop
181 states = std::vector<ALGEBRA::State<T>>();
182 times = std::vector<T>();
183 if (simparams.output_all_cycles) {
184 int num_states =
185 simparams.sim_num_time_steps / simparams.output_interval + 1;
186 states.reserve(num_states);
187 times.reserve(num_states);
188 } else {
189 int num_states =
190 simparams.sim_pts_per_cycle / simparams.output_interval + 1;
191 states.reserve(num_states);
192 times.reserve(num_states);
193 }
194 T time = 0.0;
195
196 // Run integrator
197 DEBUG_MSG("Run time integration");
198 int interval_counter = 0;
199 int start_last_cycle =
200 simparams.sim_num_time_steps - simparams.sim_pts_per_cycle;
201 if (simparams.output_all_cycles || (0 >= start_last_cycle)) {
202 times.push_back(time);
203 states.push_back(std::move(state));
204 }
205 for (int i = 1; i < simparams.sim_num_time_steps; i++) {
206 state = integrator.step(state, time);
207 interval_counter += 1;
208 time = simparams.sim_time_step_size * T(i);
209 if ((interval_counter == simparams.output_interval) ||
210 (!simparams.output_all_cycles && (i == start_last_cycle))) {
211 if (simparams.output_all_cycles || (i >= start_last_cycle)) {
212 times.push_back(time);
213 states.push_back(std::move(state));
214 }
215 interval_counter = 0;
216 }
217 }
218
219 // Make times start from 0
220 if (!simparams.output_all_cycles) {
221 T start_time = times[0];
222 for (auto& time : times) {
223 time -= start_time;
224 }
225 }
226}

233template <typename T>
234std::string Solver<T>::get_full_result() {
235 std::string output;
236 if (simparams.output_variable_based) {
237 output =
238 IO::to_variable_csv<T>(times, states, model, simparams.output_mean_only,
239 simparams.output_derivative);
240 } else {
241 output =
242 IO::to_vessel_csv<T>(times, states, model, simparams.output_mean_only,
243 simparams.output_derivative);
244 }
245 return output;
246}

248template <typename T>
249Eigen::VectorXd Solver<T>::get_single_result(std::string dof_name) {
250 int dof_index = model.dofhandler.get_variable_index(dof_name);
251 int num_states = states.size();
252 Eigen::VectorXd result = Eigen::VectorXd::Zero(num_states);
253 for (size_t i = 0; i < num_states; i++) result[i] = states[i].y[dof_index];
254 return result;
255}

257template <typename T>
258T Solver<T>::get_single_result_avg(std::string dof_name) {
259 int dof_index = model.dofhandler.get_variable_index(dof_name);
260 int num_states = states.size();
261 Eigen::VectorXd result = Eigen::VectorXd::Zero(num_states);
262 for (size_t i = 0; i < num_states; i++) result[i] = states[i].y[dof_index];
263 return result.mean();
264}

266template <typename T>
267void Solver<T>::update_block_params(std::string block_name,
268 std::vector<T> new_params) {
269 auto block = model.get_block(block_name);
270 if (new_params.size() != block->global_param_ids.size()) {
271 throw std::runtime_error(
272 "Parameter update failed! Number of provided parameters does not match "
273 "with block parameters.");
274 }
275 for (size_t i = 0; i < new_params.size(); i++) {
276 model.get_parameter(block->global_param_ids[i])->update(new_params[i]);
277 }
278}

280template <typename T>
281void Solver<T>::sanity_checks() {
282 // Check that steady initial is not used with ClosedLoopHeartAndPulmonary
283 if ((simparams.sim_steady_initial == true) &&
284 (model.get_block("CLH") != nullptr)) {
285 std::runtime_error(
286 "ERROR: Steady initial condition is not compatible with "
287 "ClosedLoopHeartAndPulmonary block.");
288 }
289}
