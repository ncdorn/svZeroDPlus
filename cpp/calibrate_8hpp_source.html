49template <typename T>
50nlohmann::json calibrate(const nlohmann::json &config) {
51 auto output_config = nlohmann::json(config);
52
53 // Read calibration parameters
54 DEBUG_MSG("Parse calibration parameters");
55 auto const &calibration_parameters = config["calibration_parameters"];
56 T gradient_tol = calibration_parameters.value("tolerance_gradient", 1e-5);
57 T increment_tol = calibration_parameters.value("tolerance_increment", 1e-10);
58 int max_iter = calibration_parameters.value("maximum_iterations", 100);
59 bool calibrate_stenosis =
60 calibration_parameters.value("calibrate_stenosis_coefficient", true);
61 bool zero_capacitance =
62 calibration_parameters.value("set_capacitance_to_zero", false);
63 T lambda0 = calibration_parameters.value("initial_damping_factor", 1.0);
64
65 int num_params = 3;
66 if (calibrate_stenosis) {
67 num_params = 4;
68 }
69
70 // Setup model
71 auto model = MODEL::Model<T>();
72 std::vector<std::tuple<std::string, std::string>> connections;
73 std::vector<std::tuple<std::string, std::string>> inlet_connections;
74 std::vector<std::tuple<std::string, std::string>> outlet_connections;
75
76 // Create vessels
77 DEBUG_MSG("Load vessels");
78 std::map<std::int64_t, std::string> vessel_id_map;
79 int param_counter = 0;
80 for (auto const &vessel_config : config["vessels"]) {
81 std::string vessel_name = vessel_config["vessel_name"];
82
83 // Create parameter IDs
84 std::vector<int> param_ids;
85 for (size_t k = 0; k < num_params; k++)
86 param_ids.push_back(param_counter++);
87 model.add_block(MODEL::BlockType::BLOODVESSEL, param_ids, vessel_name);
88 vessel_id_map.insert({vessel_config["vessel_id"], vessel_name});
89 DEBUG_MSG("Created vessel " << vessel_name);
90
91 // Read connected boundary conditions
92 if (vessel_config.contains("boundary_conditions")) {
93 auto const &vessel_bc_config = vessel_config["boundary_conditions"];
94 if (vessel_bc_config.contains("inlet")) {
95 inlet_connections.push_back({vessel_bc_config["inlet"], vessel_name});
96 }
97 if (vessel_bc_config.contains("outlet")) {
98 outlet_connections.push_back({vessel_name, vessel_bc_config["outlet"]});
99 }
100 }
101 }
102
103 // Create junctions
104 for (auto const &junction_config : config["junctions"]) {
105 std::string junction_name = junction_config["junction_name"];
106 auto const &outlet_vessels = junction_config["outlet_vessels"];
107 int num_outlets = outlet_vessels.size();
108 if (num_outlets == 1) {
109 model.add_block(MODEL::BlockType::JUNCTION, {}, junction_name);
110 } else {
111 std::vector<int> param_ids;
112 for (size_t i = 0; i < (num_outlets * (num_params - 1)); i++)
113 param_ids.push_back(param_counter++);
114 model.add_block(MODEL::BlockType::BLOODVESSELJUNCTION, param_ids,
115 junction_name);
116 }
117 // Check for connections to inlet and outlet vessels and append to
118 // connections list
119 for (auto vessel_id : junction_config["inlet_vessels"])
120 connections.push_back({vessel_id_map[vessel_id], junction_name});
121 for (auto vessel_id : outlet_vessels)
122 connections.push_back({junction_name, vessel_id_map[vessel_id]});
123 DEBUG_MSG("Created junction " << junction_name);
124 }
125
126 // Create Connections
127 DEBUG_MSG("Created connection");
128 for (auto &connection : connections) {
129 auto ele1 = model.get_block(std::get<0>(connection));
130 auto ele2 = model.get_block(std::get<1>(connection));
131 model.add_node({ele1}, {ele2}, ele1->get_name() + ":" + ele2->get_name());
132 }
133 for (auto &connection : inlet_connections) {
134 auto ele = model.get_block(std::get<1>(connection));
135 model.add_node({}, {ele}, std::get<0>(connection) + ":" + ele->get_name());
136 }
137 for (auto &connection : outlet_connections) {
138 auto ele = model.get_block(std::get<0>(connection));
139 model.add_node({ele}, {}, ele->get_name() + ":" + std::get<1>(connection));
140 }
141
142 // Finalize model
143 model.finalize();
144
145 DEBUG_MSG("Number of parameters " << param_counter);
146
147 // Read observations
148 DEBUG_MSG("Reading observations");
149 int num_obs = 0;
150 std::vector<std::vector<T>> y_all;
151 std::vector<std::vector<T>> dy_all;
152 auto y_values = config["y"];
153 auto dy_values = config["dy"];
154 for (size_t i = 0; i < model.dofhandler.get_num_variables(); i++) {
155 std::string var_name = model.dofhandler.variables[i];
156 DEBUG_MSG("Reading observations for variable " << var_name);
157 if (!y_values.contains(var_name)) {
158 std::cout << "ERROR: Missing y observation for '" << var_name << "'"
159 << std::endl;
160 exit(1);
161 }
162 if (!dy_values.contains(var_name)) {
163 std::cout << "ERROR: Missing dy observation for '" << var_name << "'"
164 << std::endl;
165 exit(1);
166 }
167 auto y_array = y_values[var_name].get<std::vector<T>>();
168 auto dy_array = dy_values[var_name].get<std::vector<T>>();
169 num_obs = y_array.size();
170 if (i == 0) {
171 y_all.resize(num_obs);
172 dy_all.resize(num_obs);
173 }
174 for (size_t j = 0; j < num_obs; j++) {
175 y_all[j].push_back(y_array[j]);
176 dy_all[j].push_back(dy_array[j]);
177 }
178 }
179 DEBUG_MSG("Number of observations: " << num_obs);
180
181 // Setup start parameter vector
182 Eigen::Matrix<T, Eigen::Dynamic, 1> alpha =
183 Eigen::Matrix<T, Eigen::Dynamic, 1>::Zero(param_counter);
184 DEBUG_MSG("Reading initial alpha");
185 for (auto &vessel_config : output_config["vessels"]) {
186 std::string vessel_name = vessel_config["vessel_name"];
187 DEBUG_MSG("Reading initial alpha for " << vessel_name);
188 auto block = model.get_block(vessel_name);
189 alpha[block->global_param_ids[0]] =
190 vessel_config["zero_d_element_values"].value("R_poiseuille", 0.0);
191 alpha[block->global_param_ids[1]] =
192 vessel_config["zero_d_element_values"].value("C", 0.0);
193 alpha[block->global_param_ids[2]] =
194 vessel_config["zero_d_element_values"].value("L", 0.0);
195 if (num_params > 3) {
196 alpha[block->global_param_ids[3]] =
197 vessel_config["zero_d_element_values"].value("stenosis_coefficient",
198 0.0);
199 }
200 }
201 for (auto &junction_config : output_config["junctions"]) {
202 std::string junction_name = junction_config["junction_name"];
203 DEBUG_MSG("Reading initial alpha for " << junction_name);
204 auto block = model.get_block(junction_name);
205 int num_outlets = block->outlet_nodes.size();
206
207 if (num_outlets < 2) {
208 continue;
209 }
210
211 for (size_t i = 0; i < num_outlets; i++) {
212 alpha[block->global_param_ids[i]] = 0.0;
213 alpha[block->global_param_ids[i + num_outlets]] = 0.0;
214 if (num_params > 3) {
215 alpha[block->global_param_ids[i + 2 * num_outlets]] = 0.0;
216 }
217 }
218 if (junction_config["junction_type"] == "BloodVesselJunction") {
219 auto resistance = junction_config["junction_values"]["R_poiseuille"]
220 .get<std::vector<T>>();
221 auto inductance =
222 junction_config["junction_values"]["L"].get<std::vector<T>>();
223 auto stenosis_coeff =
224 junction_config["junction_values"]["stenosis_coefficient"]
225 .get<std::vector<T>>();
226 for (size_t i = 0; i < num_outlets; i++) {
227 alpha[block->global_param_ids[i]] = resistance[i];
228 alpha[block->global_param_ids[i + num_outlets]] = inductance[i];
229 if (num_params > 3) {
230 alpha[block->global_param_ids[i + 2 * num_outlets]] =
231 stenosis_coeff[i];
232 }
233 }
234 }
235 }
236
237 // Run optimization
238 DEBUG_MSG("Start optimization");
239 auto lm_alg =
240 OPT::LevenbergMarquardtOptimizer(&model, num_obs, param_counter, lambda0,
241 gradient_tol, increment_tol, max_iter);
242
243 alpha = lm_alg.run(alpha, y_all, dy_all);
244
245 // Write optimized simulation config file
246 for (auto &vessel_config : output_config["vessels"]) {
247 std::string vessel_name = vessel_config["vessel_name"];
248 auto block = model.get_block(vessel_name);
249 T stenosis_coeff = 0.0;
250 if (num_params > 3) {
251 stenosis_coeff = alpha[block->global_param_ids[3]];
252 }
253 T c_value = 0.0;
254 if (!zero_capacitance) {
255 c_value = alpha[block->global_param_ids[1]];
256 }
257 vessel_config["zero_d_element_values"] = {
258 {"R_poiseuille", alpha[block->global_param_ids[0]]},
259 {"C", std::max(c_value, 0.0)},
260 {"L", std::max(alpha[block->global_param_ids[2]], 0.0)},
261 {"stenosis_coefficient", stenosis_coeff}};
262 }
263 for (auto &junction_config : output_config["junctions"]) {
264 std::string junction_name = junction_config["junction_name"];
265 auto block = model.get_block(junction_name);
266 int num_outlets = block->outlet_nodes.size();
267
268 if (num_outlets < 2) {
269 continue;
270 }
271
272 std::vector<T> r_values;
273 for (size_t i = 0; i < num_outlets; i++) {
274 r_values.push_back(alpha[block->global_param_ids[i]]);
275 }
276 std::vector<T> l_values;
277 for (size_t i = 0; i < num_outlets; i++) {
278 l_values.push_back(
279 std::max(alpha[block->global_param_ids[i + num_outlets]], 0.0));
280 }
281 std::vector<T> ste_values;
282 if (num_params > 3) {
283 for (size_t i = 0; i < num_outlets; i++) {
284 ste_values.push_back(
285 alpha[block->global_param_ids[i + 2 * num_outlets]]);
286 }
287 } else {
288 for (size_t i = 0; i < num_outlets; i++) {
289 ste_values.push_back(0.0);
290 }
291 }
292 junction_config["junction_type"] = "BloodVesselJunction";
293 junction_config["junction_values"] = {{"R_poiseuille", r_values},
294 {"L", l_values},
295 {"stenosis_coefficient", ste_values}};
296 }
297
298 output_config.erase("y");
299 output_config.erase("dy");
300 output_config.erase("calibration_parameters");
301
302 return output_config;
303}
