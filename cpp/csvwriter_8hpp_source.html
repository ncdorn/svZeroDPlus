47/**
48 * @brief Write results vessel based.
49 *
50 * @tparam T Scalar type (e.g. `float`, `double`)
51 * @param times Sequence of time steps corresponding to the solutions
52 * @param states Sequence of states corresponding to the time steps
53 * @param model The underlying model
54 * @param mean Toggle whether only the mean over all time steps should be
55 * written
56 * @return CSV encoded output string
57 */
58template <typename T>
59std::string to_vessel_csv(std::vector<T> &times,
60 std::vector<ALGEBRA::State<T>> &states,
61 MODEL::Model<T> &model, bool mean = false,
62 bool derivative = false) {
63 // Create string stream to buffer output
64 std::stringstream out;
65
66 // Create short and long buffer for lines
67 char sbuff[140];
68 char lbuff[236];
69
70 // Write column labels
71 if (derivative) {
72 out << "name,time,flow_in,flow_out,pressure_in,pressure_out,d_flow_in,d_"
73 "flow_out,d_pressure_in,d_pressure_out\n";
74 } else {
75 out << "name,time,flow_in,flow_out,pressure_in,pressure_out\n";
76 }
77
78 // Determine number of time steps
79 int num_steps = times.size();
80
81 unsigned int inflow_dof;
82 unsigned int outflow_dof;
83 unsigned int inpres_dof;
84 unsigned int outpres_dof;
85 for (size_t i = 0; i < model.get_num_blocks(); i++) {
86 auto block = model.get_block(i);
87 // Extract global solution indices of the block
88
89 if (dynamic_cast<const MODEL::BloodVessel<T> *>(block) == nullptr) {
90 continue;
91 }
92
93 std::string name = block->get_name();
94 inflow_dof = block->inlet_nodes[0]->flow_dof;
95 outflow_dof = block->outlet_nodes[0]->flow_dof;
96 inpres_dof = block->inlet_nodes[0]->pres_dof;
97 outpres_dof = block->outlet_nodes[0]->pres_dof;
98
99 // Write the solution of the block to the output file
100 if (derivative) {
101 if (mean) {
102 T inflow_mean = 0.0;
103 T outflow_mean = 0.0;
104 T inpres_mean = 0.0;
105 T outpres_mean = 0.0;
106 T d_inflow_mean = 0.0;
107 T d_outflow_mean = 0.0;
108 T d_inpres_mean = 0.0;
109 T d_outpres_mean = 0.0;
110 for (size_t i = 0; i < num_steps; i++) {
111 inflow_mean += states[i].y[inflow_dof];
112 outflow_mean += states[i].y[outflow_dof];
113 inpres_mean += states[i].y[inpres_dof];
114 outpres_mean += states[i].y[outpres_dof];
115 d_inflow_mean += states[i].ydot[inflow_dof];
116 d_outflow_mean += states[i].ydot[outflow_dof];
117 d_inpres_mean += states[i].ydot[inpres_dof];
118 d_outpres_mean += states[i].ydot[outpres_dof];
119 }
120 inflow_mean /= num_steps;
121 outflow_mean /= num_steps;
122 inpres_mean /= num_steps;
123 outpres_mean /= num_steps;
124 d_inflow_mean /= num_steps;
125 d_outflow_mean /= num_steps;
126 d_inpres_mean /= num_steps;
127 d_outpres_mean /= num_steps;
128 snprintf(
129 lbuff, 236, "%s,,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e\n",
130 name.c_str(), inflow_mean, outflow_mean, inpres_mean, outpres_mean,
131 d_inflow_mean, d_outflow_mean, d_inpres_mean, d_outpres_mean);
132 out << lbuff;
133 } else {
134 for (size_t i = 0; i < num_steps; i++) {
135 snprintf(lbuff, 236,
136 "%s,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e,%.16e\n",
137 name.c_str(), times[i], states[i].y[inflow_dof],
138 states[i].y[outflow_dof], states[i].y[inpres_dof],
139 states[i].y[outpres_dof], states[i].ydot[inflow_dof],
140 states[i].ydot[outflow_dof], states[i].ydot[inpres_dof],
141 states[i].ydot[outpres_dof]);
142 out << lbuff;
143 }
144 }
145 } else {
146 if (mean) {
147 T inflow_mean = 0.0;
148 T outflow_mean = 0.0;
149 T inpres_mean = 0.0;
150 T outpres_mean = 0.0;
151 for (size_t i = 0; i < num_steps; i++) {
152 inflow_mean += states[i].y[inflow_dof];
153 outflow_mean += states[i].y[outflow_dof];
154 inpres_mean += states[i].y[inpres_dof];
155 outpres_mean += states[i].y[outpres_dof];
156 }
157 inflow_mean /= num_steps;
158 outflow_mean /= num_steps;
159 inpres_mean /= num_steps;
160 outpres_mean /= num_steps;
161 snprintf(sbuff, 140, "%s,,%.16e,%.16e,%.16e,%.16e\n", name.c_str(),
162 inflow_mean, outflow_mean, inpres_mean, outpres_mean);
163 out << sbuff;
164 } else {
165 for (size_t i = 0; i < num_steps; i++) {
166 snprintf(sbuff, 140, "%s,%.16e,%.16e,%.16e,%.16e,%.16e\n",
167 name.c_str(), times[i], states[i].y[inflow_dof],
168 states[i].y[outflow_dof], states[i].y[inpres_dof],
169 states[i].y[outpres_dof]);
170 out << sbuff;
171 }
172 }
173 }
174 }
175
176 return out.str();
177}

179/**
180 * @brief Write results variable based.
181 *
182 * @tparam T Scalar type (e.g. `float`, `double`)
183 * @param times Sequence of time steps corresponding to the solutions
184 * @param states Sequence of states corresponding to the time steps
185 * @param model The underlying model
186 * @param mean Toggle whether only the mean over all time steps should be
187 * written
188 * @return CSV encoded output string
189 */
190template <typename T>
191std::string to_variable_csv(std::vector<T> &times,
192 std::vector<ALGEBRA::State<T>> &states,
193 MODEL::Model<T> &model, bool mean = false,
194 bool derivative = false) {
195 // Create string stream to buffer output
196 std::stringstream out;
197
198 // Create short and long buffer for lines
199 char sbuff[87];
200 char lbuff[110];
201
202 // Determine number of time steps
203 int num_steps = times.size();
204
205 // Write column labels
206 if (derivative) {
207 out << "name,time,y,ydot\n";
208 if (mean) {
209 for (size_t i = 0; i < model.dofhandler.size(); i++) {
210 std::string name = model.dofhandler.variables[i];
211 T mean_y = 0.0;
212 T mean_ydot = 0.0;
213 for (size_t j = 0; j < num_steps; j++) {
214 mean_y += states[j].y[i];
215 mean_ydot += states[j].ydot[i];
216 }
217 mean_y /= num_steps;
218 mean_ydot /= num_steps;
219 snprintf(lbuff, 110, "%s,,%.16e,%.16e\n", name.c_str(), mean_y,
220 mean_ydot);
221 out << lbuff;
222 }
223 } else {
224 for (size_t i = 0; i < model.dofhandler.size(); i++) {
225 std::string name = model.dofhandler.variables[i];
226 for (size_t j = 0; j < num_steps; j++) {
227 snprintf(lbuff, 110, "%s,%.16e,%.16e,%.16e\n", name.c_str(), times[j],
228 states[j].y[i], states[j].ydot[i]);
229 out << lbuff;
230 }
231 }
232 }
233 } else {
234 out << "name,time,y\n";
235 if (mean) {
236 for (size_t i = 0; i < model.dofhandler.size(); i++) {
237 std::string name = model.dofhandler.variables[i];
238 T mean_y = 0.0;
239 for (size_t j = 0; j < num_steps; j++) {
240 mean_y += states[j].y[i];
241 }
242 mean_y /= num_steps;
243 snprintf(sbuff, 87, "%s,,%.16e\n", name.c_str(), mean_y);
244 out << sbuff;
245 }
246 } else {
247 for (size_t i = 0; i < model.dofhandler.size(); i++) {
248 std::string name = model.dofhandler.variables[i];
249 for (size_t j = 0; j < num_steps; j++) {
250 snprintf(sbuff, 87, "%s,%.16e,%.16e\n", name.c_str(), times[j],
251 states[j].y[i]);
252 out << sbuff;
253 }
254 }
255 }
256 }
257
258 return out.str();
259}
